# Write your own code
currencies_df['bool_cond'] = (currencies_df['growth_euro'] > 2) | (currencies_df['growth_euro'] < -2)

plt.figure(figsize=(10, 6))
plt.plot(currencies_df['date'], currencies_df['growth_euro'], label='Growth Euro', color='blue')

plt.fill_between(currencies_df['date'], currencies_df['growth_euro'], where=currencies_df['bool_cond'], 
                color='red', alpha=0.3, label='Growth > 2% or < -2%')

plt.xlabel('Date')
plt.ylabel('Growth Rate (%)')
plt.title("Daily Growth Rate of Euro to USD Exchange Rate")
plt.legend()
plt.show()
